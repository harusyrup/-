# 20.有效的括号
```cpp
class Solution {
public:
    bool isValid(string s) {
        stack<char> stack;
        if(s.size()%2!=0) return false;
        for(int i = 0; i<s.size();i++){
            if(s[i]=='(') stack.push(')');
            else if(s[i]=='[') stack.push(']');
            else if(s[i]=='{') stack.push('}');
            else if(stack.empty()||s[i]!=stack.top()){
                return false;
            }else if(s[i]==stack.top()){
                stack.pop();
            }
        }
        if(stack.empty()) return true;
        else return false;
    }
};
```
